Declare @yearName int ;

set @yearName = 2024
Declare @lastdayYear datetime ;
Declare @maxDateItemClose datetime ;

Declare @totalcostAmountOfsoldItems money ;
Declare @totalcostAmountOfReturnedItems money ;

Select @lastdayYear=CAST( Convert(varchar, '31-dec-'+ CONVERT(varchar, @yearName))  as date);


Select @maxDateItemClose=ISNULL( MAX(ToDate),'01-Jan-1900') from  PurchaseItemCloses  where ToDate<=@lastdayYear ;





WITH CTE_Name_Items AS (

Select A.CompanyProductID , ISNULL( B.AvgAmount,0) AvgAmount ,  0 as TotalPurQty,0 TotalPurAmount,0 TotalPurNetPurAmount

from CompanyProducts  A left outer join PurchaseItemCloses B  on A.CompanyProductID=B.CompanyProductID  and B.ToDate=@maxDateItemClose


)
,

 CTE_Sales AS (


SELECT    A.CompanyProductID,
ISNULL(  SUM( A.SalesQtyInPair*( A.SalesRate- A.CommissionRate)),0) as SalesAmount,ISNULL(SUM( A.SalesQtyInPair),0) SalesQtyInPair



FROM            dbo.SalesDetailss AS A INNER JOIN
                         dbo.SalesHeads AS B ON A.SalesHeadID = B.SalesHeadID 
                          where datepart(year, SalesDate )=@yearName
group by  A.CompanyProductID


)


,

 CTE_Returns AS (



   Select B.CompanyProductID, SUM(B.ReyurnQtyInPair*(B.RetRate-B.ReturnCommissionRate)) ReturnAmount,SUM(B.ReyurnQtyInPair) ReyurnQtyInPair

FROM            dbo.SalesReturns AS A INNER JOIN
                         dbo.SalesReturnDetails AS B ON A.SalesReturnID = B.SalesReturnID    where   


       datepart(year, ReturnDate )=@yearName


group by  B.CompanyProductID


) 




    SELECT 
        A.CompanyProductID,A.SalesAmount / A.SalesQtyInPair  salesrate,
      case when X.AvgPurAmnt =0 then A.SalesAmount / NULLIF(A.SalesQtyInPair, 0)  else  X.AvgPurAmnt end AS AvgPurAmnt
		
		
		, A.SalesQtyInPair -- Avoid division by zero
    FROM CTE_Sales A
    LEFT JOIN (
        SELECT 
            A.CompanyProductID, 
            ISNULL(B.AvgPurAmnt, A.AvgAmount) AS AvgPurAmnt
        FROM CTE_Name_Items A
        LEFT JOIN (
            SELECT 
                B.CompanyProductID,
                SUM((B.PurchaseRate - B.CommissionRate) * B.PurchaseQtyInPair) / 
                SUM(B.PurchaseQtyInPair) AS AvgPurAmnt
            FROM  [PurchaseHeads] A
            INNER JOIN PurchaseDetails B 
                ON A.PurchaseHeadID = B.PurchaseHeadID
            WHERE CAST(A.PurchaseDate AS DATE) > @maxDateItemClose
              AND CAST(A.PurchaseDate AS DATE) <= @lastdayYear
            GROUP BY B.CompanyProductID
        ) B ON A.CompanyProductID = B.CompanyProductID
    ) X ON A.CompanyProductID = X.CompanyProductID

;
