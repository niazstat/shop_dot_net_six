
CREATE PROCEDURE SP_Customer_Close_SizeDetails(  @yearName int,@custID int)
AS
BEGIN

Select D.ArticleID,D.Name As ArticleName,E.SizeID,E.Name As SizeName,B.SalesQtyInPair from SalesHeads A inner join SalesDetailss B on A.SalesHeadID=B.SalesHeadID 

inner join CompanyProducts C on B.CompanyProductID=C.CompanyProductID

inner join Articles D on C.ArticleID=D.ArticleID inner join Size E on c.SizeID=E.SizeID

where A.GeneratedSalesNo  in (

SELECT 
AutoNo
   

  FROM [OrderManagementDB].[dbo].[YearCloseCustomerDetailses]


  where Type='Sales' and CustomerID=@custID and DATEPART(YEAR,YearCloseDate)=@yearName

  )


END
